        function initializeGlobalTimerSystem() {
            let totalSeconds = 24 * 60 * 60;
            const hourNodes = document.querySelectorAll(".hourBox");
            const minNodes = document.querySelectorAll(".minBox");
            const secNodes = document.querySelectorAll(".secBox");

            setInterval(() => {
                if (totalSeconds <= 0) { totalSeconds = 24 * 60 * 60; }
                totalSeconds--;

                let currentHrs = Math.floor(totalSeconds / 3600);
                let currentMins = Math.floor((totalSeconds % 3600) / 60);
                let currentSecs = totalSeconds % 60;

                const formattedHrs = currentHrs < 10 ? "0" + currentHrs : currentHrs;
                const formattedMins = currentMins < 10 ? "0" + currentMins : currentMins;
                const formattedSecs = currentSecs < 10 ? "0" + currentSecs : currentSecs;

                // Update all countdown elements simultaneously
                hourNodes.forEach(node => node.innerText = formattedHrs);
                minNodes.forEach(node => node.innerText = formattedMins);
                secNodes.forEach(node => node.innerText = formattedSecs);
            }, 1000);
        }
