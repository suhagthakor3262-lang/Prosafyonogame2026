            let query = document.getElementById('gameSearchEngine').value.toLowerCase();
            let cards = document.querySelectorAll('.grid-item-card');
            cards.forEach(card => {
                let text = card.querySelector('.app-label-text').textContent.toLowerCase();
                card.style.display = text.includes(query) ? "flex" : "none";
            });
        }
